#The filtering command for this dataset follows the filtering technique used in Barria et al. 2018
#The following command was entered
plink --file RISG_uf --allow-extra-chr --chr-set 94 --remove FailedQC.remove --maf 0.05 --geno 0.05 --mind 0.05 --hwe 1e-6 --recode --out RISG_f
#Arguments fed
--chr-set 94: Necessary to deal with invalid chromosome
--remove FailedQC.remove: Removed samples that were identified as poor performing by Axiom
--maf 0.05: Removed SNPs with a minor allele frequency less than 0.05
--geno 0.05: Removed SNPs with more than 5% missinng data
--mind 0.05: Removed individuals with more than 5% missing data  
--hwe 1e-6: Removed SNPs out of Hardy-Weinberg Equilibrium

#Filtering results, taken from RISG_f.log
--remove: Removed 3 individuals
--mind: Removed 6 individuals
--geno: Removed 3770 SNPs
--hwe: Removed 2389 SNPs
--maf: Removed 7578 SNPs
